317  std::conditional_t<std::is_same_v<U, void>, T, U> dot(
318  const Vector<N2, T2> &rhs) const {
319  static_assert(N == N2, "Vector size mismatch");
320  if constexpr (!std::is_same_v<T, T2>) {

329  }

333  template <typename U = void, std::size_t N2, std::size_t M2, typename T2>
334  std::conditional_t<std::is_same_v<U, void>, Vector<M2, T>, Vector<M2, U>> dot(
335  const Matrix<N2, M2, T2> &rhs) const {
336  static_assert(N == N2,
337  "Vector-Matrix dimension mismatch. Vector size must be equal "
338  "to the first dimension of the matrix");
339  if constexpr (!std::is_same_v<T, T2>) {
340  static_assert(
341  !std::is_same_v<U, void>,
342  "Vector and Matrix types are different. please specify the return "
343  "type. e.g. \"vec.dot<float>(mat);\"");
344  }
345  using U_ = std::conditional_t<std::is_same_v<U, void>, T, U>;
346  Vector<M2, U_> ret;
347  for (std::size_t i = 0; i < M2; i++) {
348  U_ sum{0};
349  for (std::size_t k = 0; k < N; k++) {
350  sum += (data_[k] * rhs[k][i]);
351  }
352  ret[i] = sum;
353  }
354  return ret;
355  }
