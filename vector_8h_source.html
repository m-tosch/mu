313  std::conditional_t<std::is_same_v<U, void>, T, U> dot(
314  const Vector<N2, T2> &rhs) const {
315  static_assert(N == N2, "Vector size mismatch");
316  if constexpr (!std::is_same_v<T, T2>) {
317  static_assert(!std::is_same_v<U, void>,
318  "Vector types are different. please specify the return "

320  }

322  U_ ret{};

324  ret += data_[i] * rhs[i];
325  }
326  return ret;

329  template <typename U = void, std::size_t N2, std::size_t M2, typename T2>
330  std::conditional_t<std::is_same_v<U, void>, Vector<M2, T>, Vector<M2, U>> dot(
331  const Matrix<N2, M2, T2> &rhs) const {
332  static_assert(N == N2,
333  "Vector-Matrix dimension mismatch. Vector size must be equal "
334  "to the first dimension of the matrix");
335  if constexpr (!std::is_same_v<T, T2>) {
336  static_assert(
337  !std::is_same_v<U, void>,
338  "Vector and Matrix types are different. please specify the return "
339  "type. e.g. \"vec.dot<float>(mat);\"");
340  }
341  using U_ = std::conditional_t<std::is_same_v<U, void>, T, U>;
342  Vector<M2, U_> ret;
343  for (std::size_t i = 0; i < M2; i++) {
344  U_ sum{0};
345  for (std::size_t k = 0; k < N; k++) {
346  sum += (data_[k] * rhs[k][i]);
347  }
348  ret[i] = sum;
349  }
350  return ret;
351  }
