24  auto start = static_cast<value_type>(0);
25  auto incr = static_cast<value_type>(1.5F); // NOLINT
26  /* put values into the test values "matrix" (array of array) */
27  for (auto& inner : values) { // rows
28  for (auto& item : inner) { // cols
29  item = (start += incr);
30  }
31  }
32  }

34  static inline T dummy;
35  /* test values. exact dimensions of the corresponding matrix-under-test */
36  std::array<mu::Vector<dummy.size()[1], value_type>, dummy.size()[0]> values;
37  /* some tests need a converted type, defined as
38  * - floating point type -> int
39  * - integral type -> float
40  * if the test input type is neither integral nor floating point, the
41  * converted type is equivalent to the test type (i.e. not converted) */
42  using ConvertedType = std::conditional_t<
43  std::is_integral_v<value_type>, float,
44  std::conditional_t<std::is_floating_point_v<value_type>, int,
45  value_type>>;
46 };

485 TYPED_TEST_P(MatrixTypeFixture, OperatorStreamOut) {
487  TypeParam obj{this->values};
489  /* 1. read matrix ostream in a stringstream */
490  std::stringstream ss;
491  ss << obj;
492  /* 2. remove unwanted characters. leave ',' separator i */
493  std::string s = ss.str();
494  char chars[] = " []\n";
495  for (unsigned int i = 0; i < strlen(chars); ++i) { // NOLINT
496  s.erase(std::remove(s.begin(), s.end(), chars[i]), s.end()); // NOLINT
497  }
498  std::stringstream sss;
499  sss << s;
500  /* 3. put values from string(stream) in a std::vector */
501  using vtype = typename TestFixture::value_type;
502  std::vector<vtype> v;
503  vtype i{};
504  while (sss >> i) {
505  v.push_back(i);
506  if (sss.peek() == ',') {
507  sss.ignore();
508  }
509  }
511  /* compare linear layed out std::vector with nested matrix structure */
512  auto rows = this->values.size();
513  auto cols = this->values[0].size();
514  for (std::size_t i = 0; i < rows; i++) {
515  for (std::size_t j = 0; j < cols; j++) {
516  EXPECT_EQ(v[i * cols + j], this->values[i][j]);
517  }
518  }
