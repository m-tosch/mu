353 TYPED_TEST_P(MatrixTypeFixture, OperatorStreamOut) {
355  TypeParam obj{this->values};
357  /* 1. read matrix ostream in a stringstream */
358  std::stringstream ss;
359  ss << obj;
360  /* 2. remove unwanted characters. leave ',' separator i */
361  std::string s = ss.str();
362  char chars[] = " []\n";
363  for (unsigned int i = 0; i < strlen(chars); ++i) { // NOLINT
364  s.erase(std::remove(s.begin(), s.end(), chars[i]), s.end()); // NOLINT
365  }
366  std::stringstream sss;
367  sss << s;
368  /* 3. put values from string(stream) in a std::vector */
369  using vtype = typename TestFixture::value_type;
370  std::vector<vtype> v;
371  vtype i{};
372  while (sss >> i) {
373  v.push_back(i);
374  if (sss.peek() == ',') {
375  sss.ignore();
376  }
377  }
379  /* compare linear layed out std::vector with nested matrix structure */
380  auto rows = this->values.size();
381  auto cols = this->values[0].size();
382  for (std::size_t i = 0; i < rows; i++) {
383  for (std::size_t j = 0; j < cols; j++) {
384  EXPECT_EQ(v[i * cols + j], this->values[i][j]);
385  }
386  }
