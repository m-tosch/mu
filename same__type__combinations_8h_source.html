42 /* test fixture for same type tests (e.g. vector<>vector). multiple
43  * inheritance, because the types inside e.g. two vectors may be different.
44  * therefore different type test values are needed. it is assumed that the
45  * sizes match */
46 template <typename T>
47 class SameTypeCombinationsFixture : public BaseTypeFixture<T, 0>,
48  public SecondBaseTypeFixture<T> {
49  /* ...the test fixture needs to know which base type fixture to use for the
50  * second test setup. this evaluates the index for that 2nd base (0 or 1) */
51  static constexpr auto base_idx() {
52  if constexpr (std::is_same_v<SecondBaseTypeFixture<T>, EmptyBase>) {
53  return std::pair<std::size_t, std::size_t>(0, 0);
54  } else {
55  return std::pair<std::size_t, std::size_t>(0, 1);
56  }
57  }
58 
59  /* actual gtest setup */
60  public:
61  using BaseTypeFixture1 = BaseTypeFixture<T, base_idx().first>;
62  using BaseTypeFixture2 = BaseTypeFixture<T, base_idx().second>;
63 
64  void SetUp() override { // NOLINT
65  BaseTypeFixture1::SetUp();
66  BaseTypeFixture2::SetUp();
67  }
68  /* "alias" variables and functions for easier use in test functions */
69  using T1 = typename std::tuple_element<base_idx().first, T>::type;
70  using T2 = typename std::tuple_element<base_idx().second, T>::type;
71  auto values() { return BaseTypeFixture1::values; }
72  auto values2() { return BaseTypeFixture2::values; }
73  auto dummy() { return BaseTypeFixture1::dummy; }
74 };
