430  template <typename U = void, std::size_t N2, std::size_t M2, typename T2>
431  std::conditional_t<std::is_same_v<U, void>, Matrix<N, M2, T>,
432  Matrix<N, M2, U>>
433  dot(const Matrix<N2, M2, T2> &rhs) const {
434  static_assert(
435  M == N2,
436  "Matrix dimension mismatch. Second dimension of first matrix must be "
437  "equal to the first dimension of the second matrix");
438  if constexpr (!std::is_same_v<T, T2>) {
439  static_assert(!std::is_same_v<U, void>,
440  "Matrix types are different. please specify the return "
441  "type. e.g. \"mat1.dot<float>(mat2);\"");
442  }
443  using U_ = std::conditional_t<std::is_same_v<U, void>, T, U>;
444  Matrix<N, M2, U_> ret;
445  for (std::size_t i = 0; i < N; i++) {
446  for (std::size_t j = 0; j < M2; j++) {
447  U_ sum{0};
448  for (std::size_t k = 0; k < M; k++) {
449  sum += (data_[i][k] * rhs[k][j]);
450  }
451  ret[i][j] = sum;
452  }
453  }
454  return ret;
455  }
