460  template <typename U = void, std::size_t N2, std::size_t M2, typename T2>
461  std::conditional_t<std::is_same<U, void>::value, Matrix<N, M2, T>,
462  Matrix<N, M2, U>>
463  dot(const Matrix<N2, M2, T2> &rhs) const {
464  static_assert(
465  M == N2,
466  "Matrix dimension mismatch. Second dimension of first matrix must be "
467  "equal to the first dimension of the second matrix");
468  using U_ = std::conditional_t<!std::is_same<T, T2>::value, U, T>;
469  static_assert(!std::is_same<U_, void>::value,
470  "Matrix types are different. please specify the return "
471  "type. e.g. \"mat1.dot<float>(mat2);\"");
472  Matrix<N, M2, U_> ret;
473  for (std::size_t i = 0; i < N; i++) {
474  for (std::size_t j = 0; j < M2; j++) {
475  U_ sum{0};
476  for (std::size_t k = 0; k < M; k++) {
477  sum += (data_[i][k] * rhs[k][j]);
478  }
479  ret[i][j] = sum;
480  }
481  }
482  return ret;
